import { ArticleLayout } from '@/app/shared/@JustLaunch/components/Blog/ArticleLayout';
import { getAllArticles } from '@/app/shared/@JustLaunch/lib/articles';

export const article = {
  author: 'Joédyson',
  date: '2024-01-04',
  title: 'Exemplo de Artigo',
  description:
    'Este é um exemplo de artigo para demonstrar como criar conteúdo usando MDX.',
};

export default async function MDXPage(props) {
  const allArticles = await getAllArticles();
  return (
    <ArticleLayout article={article} allArticles={allArticles} {...props} />
  );
}
